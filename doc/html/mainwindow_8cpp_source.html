365 void MainWindow::selectionChanged()
366 {
367  /*
368  normally, axis base line, axis tick labels and axis labels are selectable separately, but we want
369  the user only to be able to select the axis as a whole, so we tie the selected states of the tick labels
370  and the axis base line together. However, the axis label shall be selectable individually.
371 
372  The selection state of the left and right axes shall be synchronized as well as the state of the
373  bottom and top axes.
374 
375  Further, we want to synchronize the selection of the graphs with the selection state of the respective
376  legend item belonging to that graph. So the user can select a graph by either clicking on the graph itself
377  or on its legend item.
378  */
379 
380  // make top and bottom axes be selected synchronously, and handle axis and tick labels as one selectable object:
381  if (ui->chart_show->xAxis->selectedParts().testFlag(QCPAxis::spAxis) || ui->chart_show->xAxis->selectedParts().testFlag(QCPAxis::spTickLabels) ||
382  ui->chart_show->xAxis2->selectedParts().testFlag(QCPAxis::spAxis) || ui->chart_show->xAxis2->selectedParts().testFlag(QCPAxis::spTickLabels))
383  {
384  ui->chart_show->xAxis2->setSelectedParts(QCPAxis::spAxis|QCPAxis::spTickLabels);
385  ui->chart_show->xAxis->setSelectedParts(QCPAxis::spAxis|QCPAxis::spTickLabels);
386  }
387  // make left and right axes be selected synchronously, and handle axis and tick labels as one selectable object:
388  if (ui->chart_show->yAxis->selectedParts().testFlag(QCPAxis::spAxis) || ui->chart_show->yAxis->selectedParts().testFlag(QCPAxis::spTickLabels) ||
389  ui->chart_show->yAxis2->selectedParts().testFlag(QCPAxis::spAxis) || ui->chart_show->yAxis2->selectedParts().testFlag(QCPAxis::spTickLabels))
390  {
391  ui->chart_show->yAxis2->setSelectedParts(QCPAxis::spAxis|QCPAxis::spTickLabels);
392  ui->chart_show->yAxis->setSelectedParts(QCPAxis::spAxis|QCPAxis::spTickLabels);
393  }
394 
395  // synchronize selection of graphs with selection of corresponding legend items:
396  for (int i=0; i<ui->chart_show->graphCount(); ++i)
397  {
398  QCPGraph *graph = ui->chart_show->graph(i);
399  QCPPlottableLegendItem *item = ui->chart_show->legend->itemWithPlottable(graph);
400  if (item->selected() || graph->selected())
401  {
402  item->setSelected(true);
403  graph->setSelected(true);
404  }
405  }
406 }

434 void MainWindow::addRandomGraph()
435 {
436  int n = 50; // number of points in graph
437  double xScale = (rand()/(double)RAND_MAX + 0.5)*2;
438  double yScale = (rand()/(double)RAND_MAX + 0.5)*2;
439  double xOffset = (rand()/(double)RAND_MAX - 0.5)*4;
440  double yOffset = (rand()/(double)RAND_MAX - 0.5)*5;
441  double r1 = (rand()/(double)RAND_MAX - 0.5)*2;
442  double r2 = (rand()/(double)RAND_MAX - 0.5)*2;
443  double r3 = (rand()/(double)RAND_MAX - 0.5)*2;
444  double r4 = (rand()/(double)RAND_MAX - 0.5)*2;
445  QVector<double> x(n), y(n);
446  for (int i=0; i<n; i++)
447  {
448  x[i] = (i/(double)n-0.5)*10.0*xScale + xOffset;
449  y[i] = (sin(x[i]*r1*5)*sin(cos(x[i]*r2)*r4*3)+r3*cos(sin(x[i])*r4*2))*yScale + yOffset;
450  }
451 
452  ui->chart_show->addGraph();
453  ui->chart_show->graph()->setName(QString("New graph %1").arg(ui->chart_show->graphCount()-1));
454  ui->chart_show->graph()->setData(x, y);
455  ui->chart_show->graph()->setLineStyle((QCPGraph::LineStyle)(rand()%5+1));
456  if (rand()%100 > 75)
457  ui->chart_show->graph()->setScatterStyle(QCPScatterStyle((QCPScatterStyle::ScatterShape)(rand()%9+1)));
458  QPen graphPen;
459  graphPen.setColor(QColor(rand()%245+10, rand()%245+10, rand()%245+10));
460  graphPen.setWidthF(rand()/(double)RAND_MAX*2+1);
461  ui->chart_show->graph()->setPen(graphPen);
462  ui->chart_show->replot();
463 }
